import pandas as pd
import json
def get_coordinator(year):
    if 2000 <= year <= 2006:
        return 'Richt'
    elif 2007 <= year <= 2014:
        return 'Bobo'
    elif 2015 <= year <= 2015:
        return 'Schottenheimer'
    elif 2016 <= year <= 2018:
        return 'Chaney'
    elif 2019 <= year <= 2019:
        return 'Coley'
    elif 2020 <= year <= 2022:
        return 'Monken'
    else:
        return 'NA'

# get all final team rankings
def open_file(year):
    with open(f'data/get_rankings/get_rankings_{year}.json', 'r') as f:
        return [x['ranks'] for x in json.loads(f.read())[-1]['polls'] if x['poll'] == "AP Top 25"][0]
rankings_df = None
for year in range(2007, 2023):
    df_year = pd.DataFrame(open_file(year))
    df_year['year'] = year
    if rankings_df is None:
        rankings_df = df_year
    else:
        rankings_df = pd.concat([rankings_df, df_year])
rankings_df['coordinator'] = rankings_df.year.apply(get_coordinator)
