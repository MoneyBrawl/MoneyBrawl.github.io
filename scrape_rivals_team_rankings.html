def get_team_rankings_by_year(year):
    PAGE_LOAD_TIMEOUT = 60
    options = webdriver.ChromeOptions()

    options.add_argument('--headless')
    # options.headless = True
    options.add_argument('--no-sandbox')
    options.add_argument('--disable-dev-shm-usage')
    # open it, go to a website, and get results
    driver = webdriver.Chrome('/Users/bleathers/Documents/Personal/wiki_table_scrape_uga_scores/chromedriver_mac64/chromedriver', options=options)
    driver.get(f'https://n.rivals.com/team_rankings/{year}/all-teams/football')
    
    time.sleep(5)
    tag_name = 'rv-team-rankings'
    WebDriverWait(driver, PAGE_LOAD_TIMEOUT).until(EC.presence_of_element_located((By.TAG_NAME, tag_name)))
    # element = WebDriverWait(driver, PAGE_LOAD_TIMEOUT).until(EC.presence_of_element_located((By.XPATH, "//span[@class='ordinality ng-binding ng-scope' and @ng-if='category.dataType == \"rank\"' and text()='11']")))
    html = driver.page_source 
    dfs = pd.read_html(html)
    driver.close()
    return [x for x in dfs if 'POINTS' in x.columns][0]
